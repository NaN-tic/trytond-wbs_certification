Create a certification::

    >>> Certification = Model.get('wbs.certification')
    >>> certification = Certification()
    >>> certification.party = customer
    >>> certification.wbs = wbs
    >>> certification.click('set_wbs')
    >>> product_certification, service_certification = certification.lines
    >>> product_certification.quantity = 1.0
    >>> product_certification.save()
    >>> service_certification.quantity = 2.0
    >>> service_certification.save()
    >>> certification.click('proposal')
    >>> certification.state
    u'proposal'

No quantity is certified until certification is confirmed::

    >>> product_line.certified_quantity
    0.0
    >>> service_line.certified_quantity
    0.0
    >>> certification.click('confirmed')
    >>> product_line.reload()
    >>> product_line.certified_quantity
    1.0
    >>> product_line.progress
    Decimal('0.1')
    >>> service_line.reload()
    >>> service_line.certified_quantity
    2.0
    >>> service_line.progress
    Decimal('0.2')
    >>> wbs.reload()
    >>> wbs.certified_quantity
    3.0
    >>> wbs.progress
    Decimal('0.15')
